import pygame
import Box2D
from Box2D import b2World, b2PolygonShape, b2Vec2

# --- CONFIGURACIÓN ---
PPM = 20.0  # Escala: 20 píxeles = 1 metro
SCREEN_WIDTH, SCREEN_HEIGHT = 640, 480
TARGET_FPS = 60

# --- 1. MUNDO Y ESCENARIO ---
mundo = b2World(gravity=(0, -10)) # Gravedad para ver que cae

# Suelo

carga = mundo.CreateDynamicBody(position=(16, 15), angle=0.5)

# Fixture 1: El palo vertical (abajo)
carga.CreatePolygonFixture(box=(0.5, 2), density=1.0, friction=0.3)

# Fixture 2: El palo horizontal (tejado) - CORREGIDO
forma_tejado = b2PolygonShape()
# SetAsBox pide: (ancho/2, alto/2, centro_relativo, angulo)

# --- 3. PYGAME ---

    return (int(pt[0] * PPM), int(SCREEN_HEIGHT - pt[1] * PPM))
