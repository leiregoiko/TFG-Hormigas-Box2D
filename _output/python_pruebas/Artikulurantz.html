Azkeneko kodea

Hemen aurreko fitxategiko azken kodea sartuko dut. Ezaugarriak:

  • Bideoa pausatu daiteke
  • Inurri bat aukeratu daiteke ezaugarriak ikusi ahal izateko
  • Dinamika:
    • Irteera ezagutzen duten inurriek (hortik pasatu direlako nahigabe) kargara doaz feromonak uzten.
    • Irteera ezagutzen ez dutenak: feromonak jarraitzen dituzte (akatsa: irteerarantz edo kargarantz ez dute desberdintzen)
    • Kargara itsatsita daudenak: Feromonak jarraitzen dituzte irteera ez badute ezagutzen, edo ezagutzen badute irteerarantz tiratzen dute, indar gehiagorekin eta feromonak uzten.
  • Parametroak:
    • LinearDamping: 6

Bastante desastre, kostatzen zaio ateratzea eta ez dituzte feromonak jarraitzen.

import Box2D
import random
import math

# --- CONFIGURACIÓN DE PANTALLA ---

PANEL_WIDTH = 300     # Zona del panel de datos (Estilo Director)
SCREEN_WIDTH = SIM_WIDTH + PANEL_WIDTH  # Ancho total: 1100 px
SCREEN_HEIGHT = 600
TARGET_FPS = 60

# Feromonas (Rejilla - limitada solo a la zona de simulación)
GRIDSZ = 15 
COLS, ROWS = SIM_WIDTH // GRIDSZ, SCREEN_HEIGHT // GRIDSZ
mapa_feromonas = [[0.0 for _ in range(ROWS)] for _ in range(COLS)]
DECAY = 0.995 

pygame.init()
pygame.font.init()  # Inicializar el sistema de fuentes
fuente_titulo = pygame.font.SysFont("Arial", 18, bold=True)
fuente_texto = pygame.font.SysFont("Courier New", 14)

pantalla = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
reloj = pygame.time.Clock()

    # Mantiene la física dentro de los 800px de la zona de simulación
    return (int(pt[0] * PPM), int(SCREEN_HEIGHT - pt[1] * PPM))

mundo = b2World(gravity=(0, 0))

# Paredes y Salida (Ajustadas al límite de los 800px, x = 40 metros)
paredes = mundo.CreateStaticBody(position=(0, 0))
paredes.CreatePolygonFixture(box=(40, 1, (20, 1), 0))
paredes.CreatePolygonFixture(box=(40, 1, (20, 29), 0))
# El hueco de salida (en x=32 metros, unos 640px)
paredes.CreatePolygonFixture(box=(0.5, 8, (32, 25), 0)) 
paredes.CreatePolygonFixture(box=(0.5, 8, (32, 5), 0))
objetivo_salida = Box2D.b2Vec2(34, 15)

# Carga en forma de T
carga = mundo.CreateDynamicBody(position=(10, 15))
carga.CreatePolygonFixture(box=(0.5, 2.0), density=1.0)
carga.CreatePolygonFixture(box=(2.0, 0.5, (0, 2.0), 0), density=1.0)

# Hormigas
hormigas = []
uniones = {} 

for i in range(35):
    h = mundo.CreateDynamicBody(position=(random.uniform(2, 40), random.uniform(10, 40)), linearDamping=6.0)  
    h.CreateCircleFixture(radius=0.2, density=1.0)
    h.userData = {'timer_random': 0, 'dir_random': Box2D.b2Vec2(0,0), 'sabe_camino': False, 'timer_memoria': 0}
    hormigas.append(h)

def get_grid_coords(pos):
    gx = int((pos.x * PPM) // GRIDSZ)
    gy = int((SCREEN_HEIGHT - (pos.y * PPM)) // GRIDSZ)
    return max(0, min(COLS-1, gx)), max(0, min(ROWS-1, gy))

# --- VARIABLES DE CONTROL INTERACTIVO ---
pausado = False
hormiga_seleccionada = None

# --- BUCLE PRINCIPAL ---
ejecutando = True
while ejecutando:
    for evento in pygame.event.get():
        if evento.type == pygame.QUIT: 
            ejecutando = False
        
        # 1. Pausa (Barra Espaciadora)
        elif evento.type == pygame.KEYDOWN:
            if evento.key == pygame.K_SPACE:
                pausado = not pausado

        # 2. Selección con click izquierdo
        elif evento.type == pygame.MOUSEBUTTONDOWN:
            if evento.button == 1: 
                pos_raton_px = pygame.mouse.get_pos()
                # Solo procesar clicks si ocurren DENTRO de la simulación (primeros 800px)
                if pos_raton_px[0] < SIM_WIDTH:
                    raton_x = pos_raton_px[0] / PPM
                    raton_y = (SCREEN_HEIGHT - pos_raton_px[1]) / PPM
                    pos_raton_mundo = Box2D.b2Vec2(raton_x, raton_y)
                    
                    hubo_seleccion = False
                    for h in hormigas:
                        distancia = (h.position - pos_raton_mundo).length
                        if distancia < 0.6: 
                            hormiga_seleccionada = h
                            hubo_seleccion = True
                            break
                    if not hubo_seleccion:
                        hormiga_seleccionada = None

    # Fondo general (Oscuro)
    pantalla.fill((30, 30, 30))

    # 1. FEROMONAS (Dibujo y evaporación)
    for x in range(COLS):
        for y in range(ROWS):
            if mapa_feromonas[x][y] > 0.01:
                intensity = min(255, int(mapa_feromonas[x][y] * 20))
                pygame.draw.rect(pantalla, (0, intensity, intensity), (x*GRIDSZ, y*GRIDSZ, GRIDSZ, GRIDSZ))
            if not pausado:
                mapa_feromonas[x][y] *= DECAY

    # 2. LÓGICA DE COMPORTAMIENTO
    for h in hormigas:

    
        gx, gy = get_grid_coords(h.position)
        
        if not pausado:

            if h.userData['timer_memoria']==0:
                h.userData['sabe_camino'] = False
            if h.userData['timer_memoria'] > 0:
                h.userData['timer_memoria'] -= 1

            if (h.position - objetivo_salida).length < 2.0:
                h.userData['sabe_camino'] = True
                h.userData['timer_memoria'] = 480

            if h.userData['sabe_camino']:
                mapa_feromonas[gx][gy] += 3.0

        if h in uniones:
        #Está mordiendo
            if h.userData['sabe_camino']:
                dir_final = objetivo_salida - h.position
                mapa_feromonas[gx][gy] += 0.0
            else: #Esto es lo que hay que cambiar, que siga la direccion de la t, no las feromonas

                                    max_olor = mapa_feromonas[ix][iy]
                                    mejor_dir = Box2D.b2Vec2(ix*GRIDSZ/PPM, (SCREEN_HEIGHT-iy*GRIDSZ)/PPM) - h.position
            
                    dir_final = mejor_dir if mejor_dir else (carga.position - h.position)
            
            if not pausado:
                dir_final.Normalize()
                h.ApplyForceToCenter(dir_final * 28.0, wake=True)
            
        else:
        #No está mordiendo
            if not pausado:
                if (h.position - carga.position).length < 2.5:
                    for contacto in h.contacts:
                        if contacto.contact.touching and contacto.other == carga:
                            u_def = b2DistanceJointDef(bodyA=h, bodyB=carga, anchorA=h.worldCenter, anchorB=h.worldCenter, frequencyHz=0)
                            uniones[h] = Joint = mundo.CreateJoint(u_def)
                            break
                
                if h.userData['sabe_camino']:
                    #Esto se puede cambiar. Voy a poner que vaya a la carga aunque no es real.
                    dir_explora = carga.position - h.position
                else:
                    dir_explora = h.userData['dir_random']
                    if h.userData['timer_random'] <= 0:
                        ang = random.uniform(0, 6.28)
                        h.userData['dir_random'] = Box2D.b2Vec2(math.cos(ang), math.sin(ang))
                        h.userData['timer_random'] = 40
                    h.userData['timer_random'] -= 1

                dir_explora.Normalize()
                h.ApplyForceToCenter(dir_explora * 8.0, wake=True)

    # 3. AVANZAR FÍSICA
    if not pausado:
        mundo.Step(1.0/TARGET_FPS, 6, 2)

        for fixture in body.fixtures:
            shape = fixture.shape
            if isinstance(shape, b2PolygonShape):
                verts = [to_pygame(body.transform * v) for v in shape.vertices]
                pygame.draw.polygon(pantalla, (100, 150, 250), verts)
            elif isinstance(shape, b2CircleShape):
                if body == hormiga_seleccionada:
                    color = (255, 0, 0)
                else:
                    if body in uniones: color = (0, 100, 255)
                    elif body.userData.get('sabe_camino'): color = (0, 255, 100)
                    else: color = (200, 200, 200)
                
                pygame.draw.circle(pantalla, color, to_pygame(body.position), int(0.2 * PPM))
                
                if body == hormiga_seleccionada:
                    pygame.draw.circle(pantalla, (255, 255, 0), to_pygame(body.position), int(0.5 * PPM), 1)

    pygame.draw.circle(pantalla, (255, 215, 0), to_pygame(objetivo_salida), 10, 2)

    # =========================================================================
    # 5. DIBUJO DEL PANEL LATERAL GRÁFICO 
    # =========================================================================
    # Pintamos una caja negra sólida a la derecha para el panel de datos
    pygame.draw.rect(pantalla, (15, 15, 15), (SIM_WIDTH, 0, PANEL_WIDTH, SCREEN_HEIGHT))
    # Línea divisoria blanca vertical

    # Mantiene la física dentro de los 800px de la zona de simulación
    return (int(pt[0] * PPM), int(SCREEN_HEIGHT - pt[1] * PPM))

paredes.CreatePolygonFixture(box=(40, 1, (20, 1), 0))
paredes.CreatePolygonFixture(box=(40, 1, (20, 29), 0))
# El hueco de salida (en x=32 metros, unos 640px)

paredes.CreatePolygonFixture(box=(0.5, 8, (32, 5), 0))

carga.CreatePolygonFixture(box=(0.5, 2.0), density=1.0)

# Hormigas

uniones = {} 

for i in range(35):
    h = mundo.CreateDynamicBody(position=(random.uniform(2, 40), random.uniform(10, 40)), linearDamping=6.0)  
    h.CreateCircleFixture(radius=0.2, density=1.0)
    h.userData = {'timer_random': 0, 'dir_random': Box2D.b2Vec2(0,0), 'sabe_camino': False, 'timer_memoria': 0}
    hormigas.append(h)

def get_grid_coords(pos):
    gx = int((pos.x * PPM) // GRIDSZ)
    gy = int((SCREEN_HEIGHT - (pos.y * PPM)) // GRIDSZ)
    return max(0, min(COLS-1, gx)), max(0, min(ROWS-1, gy))

        elif evento.type == pygame.KEYDOWN:
